dataset = []

for row in data3s.itertuples():
    y, sr = librosa.load('data/UrbanSound8K/' + row.path, duration=2.97)   # y: audio time series, sampling rate of y
    ps = librosa.feature.melspectrogram(y=y, sr=sr)                        # Mel spectogram
    if ps.shape != (128, 128): continue
    dataset.append( (ps, row.classID) )
